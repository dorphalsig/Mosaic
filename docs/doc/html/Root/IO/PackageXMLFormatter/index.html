@Operation write(object : Object,out : ElementOutputChannel):XCore::Element
  let caseObj = object then
      caseObjType = caseObj.of()
  in if caseObjType = XCore::Attribute
     then self.writeAttribute(object,out)
     elseif caseObjType = XCore::Class
     then self.writeClass(object,out)
     elseif caseObjType = XCore::Package
     then self.writePackage(object,out)
     else super(object,out)
     end 
  end
end
@Operation writeAttribute(attribute : XCore::Element,out : XCore::Element):XCore::Element
  let modifier = if attribute.hasAtomicType()
                 then "1"
                 elseif attribute.hasSeqType()
                 then "{ordered}"
                 else "*"
                 end 
  in null;
     format(out,"<~S",Seq{"Attribute"});
     format(out," ~S='~S'",Seq{"id",out.id(attribute)});
     format(out," ~S='~S'",Seq{"name",attribute.name()});
     format(out," ~S='~S'",Seq{"type",attribute.underlyingType().path()});
     format(out," ~S='~S'",Seq{"modifier",modifier});
     format(out,"/>",Seq{"Attribute"})
  end
end
@Operation writeClass(class : XCore::Element,out : XCore::Element):XCore::Element

          format(out," ~S='~S'",Seq{"parents",encoded});
          format(out,">");

            self.writeAttribute(attribute,out)
          end;

            out.writeValue(operation)
          end;

       end
  else self.writeNameSpace(class,out)
  end 
end
