Package

Overview

A package definition is represented as an instance of this class. When the instance is performed it creates a new kernel package.

@Operation pprint(out : XCore::Element,indent : XCore::Element):XCore::Element
  format(out,"@Package ~S",Seq{name});
  if meta <> null
  then

    meta.pprint(out,indent)
  else false
  end ;
  if not parents->isEmpty
  then 
    format(out," extends ");
    @For parent in parents do
      parent.pprint(out,indent);
      if not isLast
      then format(out,",")
      else false
      end 
    end
  else false
  end ;
  format(out,"~%~V",Seq{indent + 2});
  if defs.isKindOf(Seq(Element))
  then @For def in defs do
         def.pprint(out,indent + 2);

         then format(out,"~%~V",Seq{indent + 2})
         else false
         end 
       end
  else defs.pprint(out,indent)
  end ;
  format(out,"~%~Vend",Seq{indent})
