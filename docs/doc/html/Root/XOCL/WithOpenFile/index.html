WithOpenFile

Overview

Used to bind a name to an input or output file channel such that the closing of the channel is guaranteed even in the event of an exception being raised.

@Operation desugar():Performable
  [| let Drop(Var(channel,36)) = <self.openChannel()>
    in try
         let result = <body>
         in <OCL::Var(channel)>.close();
            result
         end
       catch(exception)
         <OCL::Var(channel)>.close();
         throw exception
       end
    end |]
end
@Operation openChannel():Performable

  end 
end
