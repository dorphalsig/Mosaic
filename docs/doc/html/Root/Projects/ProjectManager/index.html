ProjectManager

Overview

A project manager has a collection of projects. The manager is associated with a resource name that defines the default location for the projects. The resource associated with the project manager also contains a manifest file that lists all of the projects.

@Operation loadMosaicProject(path : String,fix : Boolean):XCore::Element
  let dialog = xmf.busy("Loading")
  in try
       let pair = self.loadProject(path,fix)
       in if pair <> null
          then let project = pair->head;
                   mappings = pair->tail;
                   clients = xmf.getAllClients()
               in if project.version() <> xmf.version()
                  then mappings := self.fixMosaicProject(project,mappings)
                  else false
                  end ;
                  @For client in clients do
                    let clientMappings = Seq{}
                    in @For mapping in mappings do
                         if client.handles(mapping.of())
                         then clientMappings := clientMappings->including(mapping)
                         else false
                         end 
                       end;
                       client.renewAll(clientMappings)
                    end
                  end;
                  dialog.finished();
                  project
               end
          else
            dialog.finished();
            null
          end 
       end
     catch(exception)
       dialog.finished();
       throw exception
     end
  end
end

@Operation loadProject(path : String,fix : Boolean):Element
  let install = true
  in if path.fileExists()
     then let ein = ElementInputChannel(GZipInputChannel(FileInputChannel(path))) then
              pair = ein.read() then
              project = pair->head then
              package = project.package;
              resourceName = path
          in if fix
             then Walkers::Fix().walk(pair,null)
             else false
             end ;
             if Root.hasElement(package.name.toString())
             then install := xmf.confirm("Overwrite " + package.name.toString() + "?")
             else false
             end ;
             if install
             then 
               if project.resourceName.subst("","/",true).subst("","\",true) <> resourceName.subst("","/",true).subst("","\",true)
               then project.resourceName := resourceName
               else false
               end ;
               project.setDaemons();
               self.add(project);
               Root.add(package);
               project.setDaemonsActive(true);
               ein.close()
             else pair := null
             end ;
             pair
          end
     elseif path <> ""
     then throw FileNotFound(path)
     else false
     end 
  end
end
