Edge

Overview

Straightens the edge by comparing a given waypoint to the next waypoint from the list waypoints, which is supposed to hold the remaining waypoints of this edge. The next waypoint is then moved on the x- or y-axis to the point where it is on the same x- or y-axis as the waypoint before, where the position on the other axis is kept. It is moved on that axis which requires less movement. After that the procedure is repeated on the remaining waypoints. If the list of remaining waypoints is empty the nex cycle will stop the recursion. This operation should not be accessed externally as it is the iterating part od a recursion. It should only be used by the counterpart operation which will do the initialisation of the recursion.

@Operation applyWayPointLayout(layout : Seq(XML::Element)):XCore::Element
  @Count index from 0 to layout->size do

    in let findSeq = wayPoints->asSeq;

           findResult = null
       in declare(hidden,findSeq);
          declare(hidden,findDone);
          declare(hidden,findResult);
          @While not findSeq->isEmpty and not findDone do
            let wayPoint = findSeq->head then
                findElement = findSeq->head
            in declare(hidden,findElement);
               if wayPoint.index = index
               then 
                 findDone := true;
                 findResult := wayPoint.move(x,y)
               else findSeq := findSeq->tail
               end 
            end
          end;
          if not findDone
          then let wayPoint = WayPoint(index,x,y)
               in self.addWayPoint(wayPoint);
                  wayPoint.new(self)
               end
          else findResult
          end 
       end;
       @For wayPoint in wayPoints->drop(layout->size) do
         wayPoint.delete()
       end
    end
  end

      p = Point(point.x,point.y)
  in if p.x < portPos.x
     then p.x := portPos.x

  if not self.hidden()
  then let style = "0"
       in if self.lineStyle = Line::dashLine
          then style := "15 5"
          else false
          end ;

          then let waypoints = wayPoints->asSeq->sort(@Operation anonymous(x : XCore::Element,y : XCore::Element):XCore::Element
                                 x.index < y.index
                               end) then
                   firstWaypoint = waypoints->at(0) then
                   lineEnd = Point(firstWaypoint.x,firstWaypoint.y) then
                   lineStart = self.findEdgeConnection(source,lineEnd)
               in self.printEdgeLabels(lineStart,"start",out);
                  @For waypoint in waypoints do
                    lineEnd := Point(waypoint.x,waypoint.y);
                    self.printSVGLine(out,lineStart,lineEnd,0,0,style);
                    lineStart := lineEnd
                  end;
                  lineEnd := self.findEdgeConnection(target,lineStart);
                  self.printSVGLine(out,lineStart,lineEnd,sourceHead,targetHead,style);
                  self.printEdgeLabels(lineEnd,"end",out)
               end

                   lineStart = self.findEdgeConnection(source,point) then
                   lineEnd = self.findEdgeConnection(target,point)
               in self.printEdgeLabels(lineStart,"start",out);
                  self.printSVGLine(out,lineStart,lineEnd,sourceHead,targetHead,style);
                  self.printEdgeLabels(lineEnd,"end",out)
               end
          end 
       end
  else false
  end
