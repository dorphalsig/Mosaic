@Operation cardLayout(maxStack : Integer,widthSep : Integer,dx : Integer,dy : Integer):XCore::Element
  let n = 0
  in @For node in graph.nodes->asSeq->reverse do
       let x1 = n.div(maxStack) * widthSep;
           x2 = (n.mod(maxStack) + 1) * dx then
           x = x1 + x2;
           y = (n.mod(maxStack) + 1) * dy
       in node.move(x,y);
          n := n + 1
       end
     end
  end
end

@Operation showMenuOperation():XCore::Element
  let nodes = self.nodes()->select(node |
                node.hidden())->asSeq;
      edges = self.edges()->select(edge |
                edge.hidden())->asSeq then
      elements = xmf.selects("Hidden Diagram Elements","Select hidden elements",nodes + edges,null,null)
  in if elements <> 0 - 1
     then @For element in elements do
            element.show()
          end
     else false
     end 
  end
end

@Operation synchronise():XCore::Element
  let id = Seq{"identity",id};
      name = Seq{"name",name};
      isOpen = Seq{"isOpen",isOpen};
      zoom = Seq{"zoom",zoom};
      nestedZoom = Seq{"nestedZoom",nestedZoom};
      x = Seq{"x",x};
      y = Seq{"y",y};
      width = Seq{"width",width};
      height = Seq{"height",height};
      hidden = Seq{"hidden",hidden};
      shown = Seq{"shown",shown} then
      attributes = Seq{id,name,isOpen,zoom,nestedZoom,x,y,width,height,hidden,shown}
  in try
       if self.hasCommandInterpreter()
       then self.commandInterpreter().send("startElement",Seq{"diagram",attributes})
       else false
       end 
     catch(exception)
       exception.edit()
     end;
     graph.synchronise();
     try
       if self.hasCommandInterpreter()
       then self.commandInterpreter().send("endElement",Seq{"diagram"})
       else false
       end 
     catch(exception)
       exception.edit()
     end
  end
end
@Operation tileLayout(width : Integer,height : Integer,maxWidth : Integer):XCore::Element
  let x = 10;
      y = 10
  in @For node in graph.nodes do
       node.move(x,y);
       x := x + width;
       if x > maxWidth
       then 
         x := 10;
         y := y + height
       else false
       end 
     end
  end
end

  let fullText = name;

     else false
     end ;
     fullText
  end
end
