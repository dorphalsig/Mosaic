@Operation move(newx : XCore::Element,newy : XCore::Element):XCore::Element
  @When newx <> x or newy <> y do
    let oldx = x;
        oldy = y;
        doMove = true
    in @For l in listeners do
         let event = PreNodeMoveEvent(self,newx,newy,true)
         in l(event);
            doMove := event.doIt;
            newx := event.x;
            newy := event.y
         end
       end;
       if doMove
       then 
         self.x := newx;
         self.y := newy;
         try
           if self.hasCommandInterpreter()
           then self.commandInterpreter().send("moveCommand",Seq{self,newx,newy})
           else false
           end 
         catch(exception)
           exception.edit()
         end;
         @For edge in self.diagram().edges() do
           if edge.sourceNode() = self
           then edge.sourceMoved(self,newx,newy,oldx,oldy)
           elseif edge.targetNode() = self
           then edge.targetMoved(self,newx,newy,oldx,oldy)
           else false
           end 
         end;
         @For l in listeners do
           let event = PostNodeMoveEvent(self)
           in l(event)
           end
         end
       else try
              if self.hasCommandInterpreter()
              then self.commandInterpreter().send("moveCommand",Seq{self,oldx,oldy})
              else false
              end 
            catch(exception)
              exception.edit()
            end
       end 
    end
  end
end

          @For display in displays do
            display.hide()
          end;
          icon.show()
       end
  elseif icon.filename() <> filename
  then 
    self.showAsNode();
    self.showAsIcon(filename)
  else false
  end 
end
@Operation showAsNode():XCore::Element
  if iconised
  then 
    @For display in displays do
      display.show()
    end;
    self.remove(icon);
    icon.delete();
    self.icon := null;
    self.iconised := false
  else false
  end 
end
@Operation synchronise():XCore::Element
  let id = Seq{"identity",id};

      iconised = Seq{"iconised",iconised};
      hidden = Seq{"hidden",hidden};
      selectable = Seq{"selectable",selectable} then
      attributes = Seq{id,x,y,width,height,iconised,hidden,selectable}
  in try
       if self.hasCommandInterpreter()
       then self.commandInterpreter().send("startElement",Seq{"node",attributes})
       else false
       end 
     catch(exception)
       exception.edit()
     end;
     @For port in ports do
       port.synchronise()
     end;
     @For display in displays do
       display.synchronise()
     end;
     try
       if self.hasCommandInterpreter()
       then self.commandInterpreter().send("endElement",Seq{"node"})
       else false
       end 
     catch(exception)
       exception.edit()
     end
  end
end

  end
