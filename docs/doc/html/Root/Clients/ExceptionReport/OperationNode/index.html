@Operation doubleSelected():XCore::Element
  let resourceName = operation.codeBox.resourceName()
  in if resourceName <> "" and resourceName <> null
     then let path1 = resourceName;
              path2 = xmf.homeDir() + "/" + resourceName;
              lineNumber = self.owner.frame.line
          in if path1.fileExists()
             then self.editText(path1,lineNumber)
             elseif path2.fileExists()
             then self.editText(path2,lineNumber)
             else xmf.errorMessage("Cannot find " + resourceName + " in xmf.home() or as absolute path.")
             end 
          end
     else false
     end 
  end
end
