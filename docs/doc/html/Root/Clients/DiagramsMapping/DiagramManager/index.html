@Operation newDiagramName(element : XCore::Element,useDialog : XCore::Element):String

                 diagrams = self.diagramsFor(element)
             in @While let findSeq = diagrams->asSeq;
                    findDone = false;
                    findResult = null
                in declare(hidden,findSeq);
                   declare(hidden,findDone);
                   declare(hidden,findResult);
                   @While not findSeq->isEmpty and not findDone do
                     let d = findSeq->head then
                         findElement = findSeq->head
                     in declare(hidden,findElement);
                        if d.name = "diagram" + index.toString()
                        then 
                          findDone := true;
                          findResult := true
                        else findSeq := findSeq->tail
                        end 
                     end
                   end;
                   if not findDone
                   then false
                   else findResult
                   end 
                end do
                  index := index + 1
                end;
                "diagram" + index.toString()
             end
  in if useDialog
     then name := xmf.input("New Diagram","There are no diagrams currently available for this element." + "Enter a name for a new diagram.",name)
     else false
     end ;

  end
