     elseif caseObjType = Enum
     then let names = caseObj.names then
              name = caseObj.name then
              exps = caseObj.exps then
              line = caseObj.line
          in self.walkEnum(line,exps,name,names,arg)
          end
     elseif caseObjType = Doc
     then let doc = caseObj.doc then
              line = caseObj.line
          in self.walkDoc(line,doc,arg)
          end
     elseif caseObjType = Count
     then let var = caseObj.var then
              upper = caseObj.upper then
              lower = caseObj.lower then
              body = caseObj.body then
              line = caseObj.line
          in self.walkCount(line,body,lower,upper,var,arg)
          end
     elseif caseObjType = Constructor
     then let toString = caseObj.toString then
              names = caseObj.names then
              doc = caseObj.doc then
              body = caseObj.body then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart then
              line = caseObj.line
          in self.walkConstructor(line,charStart,charEnd,body,doc,names,toString,arg)
          end
     elseif caseObjType = Constraint
     then let reason = caseObj.reason then
              name = caseObj.name then
              body = caseObj.body then
              line = caseObj.line
          in self.walkConstraint(line,body,name,reason,arg)
          end
     elseif caseObjType = Clause
     then let patterns = caseObj.patterns then
              name = caseObj.name then
              body = caseObj.body then
              bindings = caseObj.bindings
          in self.walkClause(bindings,body,name,patterns,arg)
          end
     elseif caseObjType = Class
     then let parents = caseObj.parents then
              name = caseObj.name then
              metaClass = caseObj.metaClass then
              isAbstract = caseObj.isAbstract then
              defs = caseObj.defs then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart then
              line = caseObj.line
          in self.walkClass(line,charStart,charEnd,defs,isAbstract,metaClass,name,parents,arg)
          end
     elseif caseObjType = Character
     then let char = caseObj.char
          in self.walkCharacter(char,arg)
          end
     elseif caseObjType = CaseObj
     then let line = caseObj.line
          in self.walkCaseObj(line,arg)
          end
     elseif caseObjType = CaseIntRange
     then let upper = caseObj.upper then
              lower = caseObj.lower
          in self.walkCaseIntRange(lower,upper,arg)
          end
     elseif caseObjType = CaseIntConst
     then let value = caseObj.value
          in self.walkCaseIntConst(value,arg)
          end
     elseif caseObjType = CaseIntClause
     then let head = caseObj.head then
              body = caseObj.body
          in self.walkCaseIntClause(body,head,arg)
          end
     elseif caseObjType = CaseInt
     then let otherwise = caseObj.otherwise then
              limit = caseObj.limit then
              exp = caseObj.exp then
              clauses = caseObj.clauses then
              line = caseObj.line
          in self.walkCaseInt(line,clauses,exp,limit,otherwise,arg)
          end
     elseif caseObjType = CaseArm
     then let patterns = caseObj.patterns then
              action = caseObj.action
          in self.walkCaseArm(action,patterns,arg)
          end
     elseif caseObjType = Case
     then let values = caseObj.values then
              default = caseObj.default then
              arms = caseObj.arms then
              line = caseObj.line
          in self.walkCase(line,arms,default,values,arg)
          end
     elseif caseObjType = Bind
     then let value = caseObj.value then
              name = caseObj.name then
              line = caseObj.line
          in self.walkBind(line,name,value,arg)
          end
     elseif caseObjType = AttributeModifiers
     then let updater = caseObj.updater then
              setOwner = caseObj.setOwner then
              reducer = caseObj.reducer then
              guardExp = caseObj.guardExp then
              guard = caseObj.guard then
              extender = caseObj.extender then
              accessor = caseObj.accessor
          in self.walkAttributeModifiers(accessor,extender,guard,guardExp,reducer,setOwner,updater,arg)
          end
     elseif caseObjType = Attribute
     then let type = caseObj.type then
              properties = caseObj.properties then
              name = caseObj.name then
              mult = caseObj.mult then
              modifiers = caseObj.modifiers then
              meta = caseObj.meta then
              init = caseObj.init then
              doc = caseObj.doc then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart then
              line = caseObj.line
          in self.walkAttribute(line,charStart,charEnd,doc,init,meta,modifiers,mult,name,properties,type,arg)
          end
     elseif caseObjType = AbstractOp
     then let type = caseObj.type then
              name = caseObj.name then
              doc = caseObj.doc then
              args = caseObj.args then
              line = caseObj.line
          in self.walkAbstractOp(line,args,doc,name,type,arg)
          end
     elseif o.isKindOf(Sugar)
     then self.walk(o.desugar(),arg)
     else super(o,arg)
     end 
  end
end
