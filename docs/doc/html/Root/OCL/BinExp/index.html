  end 
end
@Operation compileBasicBinOp(env : Element,frame : Element,isLast : Element,saveSource : Element):Element

     end 
  end

@Operation desugar():Performable
  if binOp = "<>"
  then [| not (<left> = <right>) |]
  elseif binOp = ">="
  then [| <left> > <right> or <left> = <right> |]
  elseif binOp = "<="
  then [| <left> < <right> or <left> = <right> |]
  elseif binOp = "andthen"
  then [| if <left>
         then <right>
         else false
         end  |]
  elseif binOp = "orelse"
  then [| if <left>
         then true
         else <right>
         end  |]
  else self.error("Unknown binary operator in " + self.pprint())
  end 
end
@Operation eval(target : XCore::Element,env : XCore::Element,imports : XCore::Element):XCore::Element
  @Case binOp of
    [ "andthen" ] do
      [| if <left>
        then <right>
        else false
        end  |].eval(target,env,imports)
    end
    [ "orelse" ] do
      [| if <left>
        then true
        else <right>
        end  |].eval(target,env,imports)
    end
    else let v1 = left.eval(target,env,imports);
        v2 = right.eval(target,env,imports)
    in @Case binOp of
         [ "+" ] do
           v1 + v2
         end
         [ "-" ] do
           v1 - v2
         end
         [ "*" ] do
           v1 * v2
         end
         [ "/" ] do
           v1 / v2
         end
         [ "and" ] do
           v1 and v2
         end
         [ "or" ] do
           v1 or v2
         end
         [ ">" ] do
           v1 > v2
         end
         [ "<" ] do
           v1 < v2
         end
         [ "<=" ] do
           v1 <= v2
         end
         [ ">=" ] do
           v1 >= v2
         end
         [ "=" ] do
           v1 = v2
         end

           v1 <> v2
         end

           v1 implies v2
         end
         else v1.send(binOp,Seq{v2})
       end
    end
  end
end
