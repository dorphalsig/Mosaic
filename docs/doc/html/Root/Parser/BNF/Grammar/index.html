  end

               end 
            end

       end

@Operation cleanClauses():XCore::Element
  let cleanClauses = Set{}
  in @For c in clauses do
       let findSeq = self.inheritedClauses(c.name())->asSeq;
           findDone = false;
           findResult = null
       in declare(hidden,findSeq);

          declare(hidden,findResult);
