@Operation handleChangedAttribute(tool : XCore::Element,object : XCore::Element,display : XCore::Element,new : XCore::Element,old : XCore::Element):XCore::Element

       end

                                 then let name = attribute.name() then
                                          oldValue = object.get(name);

                                                       d.type() = type.displayType()) then

                                          newCollection = collection.replaceAt(index,self.coerceValue(new));
                                          name = attribute.name()
                                      in MDCSetSlot(object,name,newCollection);
                                         tool.slotChangedEvent(object,name,collection,newCollection);
                                         tool.disassociate(old,display);
                                         tool.associate(new,display)
                                      end
                                 end 
       in @For withoutEventsTool in tools do
            let withoutEventsState = stateTable.get(withoutEventsTool)
            in withoutEventsTool.setHandleEvents(withoutEventsState)
            end
          end;

       end
