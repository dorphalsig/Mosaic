              head = caseObj.head then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkConsExp(charStart,charEnd,head,tail,arg)
          end
     elseif caseObjType = Condp
     then let pattern = caseObj.pattern then
              condition = caseObj.condition
          in self.walkCondp(condition,pattern,arg)
          end
     elseif caseObjType = CollExp
     then let collection = caseObj.collection then
              collOp = caseObj.collOp then
              args = caseObj.args then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkCollExp(charStart,charEnd,args,collOp,collection,arg)
          end
     elseif caseObjType = BoolExp
     then let value = caseObj.value then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart then
              line = caseObj.line
          in self.walkBoolExp(line,charStart,charEnd,value,arg)
          end
     elseif caseObjType = Binding
     then let name = caseObj.name then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkBinding(charStart,charEnd,name,arg)
          end
     elseif caseObjType = BinExp
     then let right = caseObj.right then
              left = caseObj.left then
              binOp = caseObj.binOp then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkBinExp(charStart,charEnd,binOp,left,right,arg)
          end
     elseif caseObjType = Apply
     then let operator = caseObj.operator then
              args = caseObj.args then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkApply(charStart,charEnd,args,operator,arg)
          end
     elseif caseObjType = Addp
     then let right = caseObj.right then
              left = caseObj.left then
              charEnd = caseObj.charEnd then
              charStart = caseObj.charStart
          in self.walkAddp(charStart,charEnd,left,right,arg)
          end
     else o
     end 
  end
end
