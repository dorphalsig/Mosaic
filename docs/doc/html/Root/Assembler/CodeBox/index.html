@Operation resolveLabelsForward():XCore::Element
  let is = code->reverse
  in @While not is->isEmpty do
       let i = is->head

            Skip do

@Operation symbolOffset(symbol : XCore::Element):Element
  let findSeq = constants->asSeq;
      findDone = false;
      findResult = null
  in declare(hidden,findSeq);
     declare(hidden,findDone);
     declare(hidden,findResult);
     @While not findSeq->isEmpty and not findDone do
       let c = findSeq->head then
           findElement = findSeq->head
       in declare(hidden,findElement);
          if c.isKindOf(Compiler::Symbol) andthen c.name = symbol.name
          then 
            findDone := true;
            findResult := constants->indexOf(c)
          else findSeq := findSeq->tail
          end 
       end
     end;
     if not findDone
     then 
       self.constants := constants + Seq{symbol};
       constants->size - 1
     else findResult
     end 
  end
end
