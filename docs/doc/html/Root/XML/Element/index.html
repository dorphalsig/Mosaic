  @Case tag of

@Operation reduceObject(idTable : Table):Object
  if children->isEmpty
  then throw ExpectingTag("Ref")
  else let class = children->head.reduceRef() then
           o = class.new();
           id = self.getAtt("id")
       in idTable.put(id,o);
          @For c in children->tail do
            c.reduceSlot(o,idTable)
          end;
          o
       end
  end 
end
@Operation reduceRef():Class

            c.nameSpaceRef(e))
       end
  end 
end
