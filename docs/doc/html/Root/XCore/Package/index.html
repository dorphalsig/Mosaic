Package

Overview

XMF supports name spaces that contain collections of named elements. The XCore class Package is used to structure collections of class and sub-package definitions. XMF-Mosaic is structured as a tree of packages containing definitions of all aspects of the system (including XCore). The root name space is called Root; all XMF classes can be referenced via Root. Unlike UML and MOF, XCore packages are subclasses of Class. They can therefore be instantiated and can have operations, attributes and constraints.

@Operation checkMetaConstraints():ConstraintReport
  @Letrec 
    classify(metaPackage : Package,prev : Package,done : Set(Constraint),level : Integer):Set(ConstraintReport) = let C : Set(Constraint) = metaPackage.allConstraints() then
                 reports : Set(ConstraintReport) = runConstraints(C->reject(c |
                             done->contains(c)),level)
             in prev := metaPackage;
                metaPackage := metaPackage.metaPackage;
                if prev = metaPackage
                then reports
                else reports + classify(metaPackage,prev,done + C,level + 1)
                end 
             end;
    applyConstraint(c : Constraint,o : Object,level : Integer):Set(ConstraintReport) = @TypeCase(o)
                      Package do
                        Set{doConstraint(c,o,level)} + o.classes->collect(class |
                          applyConstraint(c,class,level)).flatten()
                      end
                      Class do
                        Set{doConstraint(c,o,level)} + o.attributes->collect(att |
                          applyConstraint(c,att,level)).flatten()
                      end
                      Attribute do
                        Set{doConstraint(c,o,level)}
                      end
                    end;
    doConstraint(c : Constraint,o : Object,level : Integer):ConstraintReport = try
                   c(o,c.owner,level)
                 catch(exception)
                   ConstraintReport(self,c,false,"Error while performing classification: " + exception.message)
                 end;
    runConstraints(C : Set(Constraint),level : Integer):Set(ConstraintReport) = let reports : Set(ConstraintReport) = Set{}
                   in @While not C->isEmpty do
                        let c = C->sel then
                            R : Set(ConstraintReport) = applyConstraint(c,self,level)
                        in reports := reports + R;
                           C := C->excluding(c)
                        end
                      end;
                      reports
                   end
  in let reports : Set(ConstraintReport) = classify(metaPackage,metaPackage,Set{},1)
    in ConstraintReport(self,null,reports->forAll(r |
         r.satisfied()),"",reports->asSeq)
    end
  end
end

          end
     else false
     end 
  end
end
@Operation writeWalker(dir : String,name : String,extras : Seq(Classifier),package : Package):XCore::Element
  let manifest = OrderedContainer()
  in manifest.add(name);
     @For c in classes->reject(c |
       c.name().toString() = name) + extras->asSet do
       let file = c.name() + ".xmf" then
           path = dir + "/" + file;
           object = "a" + c.name();
           A = if c.isKindOf(Class)
               then c.allAttributes()->asSeq
               else Seq{}
               end 
       in @WithOpenFile(fout -> path)
            format(fout,"parserImport Walkers;~%~%import ~S;~%import ~S;~%~%",Seq{self.path(),package.path()});
            format(fout,"@Handler ~S in ~S(~S,arg,seen)~%",Seq{c.name(),name,object});
            format(fout,"  if seen~%");
            format(fout,"  then ~S~%",Seq{object});
            format(fout,"  else~%");
            if A->isEmpty
            then format(fout," ~S~%",Seq{object})
            else
              format(fout,"    let ~{;~%        ~;~S~}~%",Seq{A->collect(a |
                if a.hasAtomicType()
              then formats("~S = self.dispatch(~S.~^~^~S,arg)",Seq{a.name(),object})
              else formats("~S = ~S.~^~^~S->collect(x | self.dispatch(x,arg))",Seq{a.name(),object})
              end )});
              format(fout,"    in ~S~%",Seq{object});
              format(fout,"    end~%")
            end ;
            format(fout,"  end~%");
            format(fout,"end;~%")
          end;
          manifest.add(c.name())
       end
     end;
     @WithOpenFile(fout -> dir + "/" + name + ".xmf")
       format(fout,"parserImport XOCL;~%~%import Walkers;~%~%");
       format(fout,"context ~S~%",Seq{package.path()});
       format(fout,"  @Class ~S metaclass Dispatcher~%",Seq{name});
       format(fout,"  end~%")
     end;
     @WithOpenFile(fout -> dir + "/Manifest.xmf")
       format(fout,"parserImport Manifests;~%~%@Manifest CodeWalker~%");
       @For file in manifest.contents() do
         format(fout,"  @File ~S end~%",Seq{file})
       end;
       format(fout,"end;~%")
     end
  end
end
@Operation writeWalker(file : String):XCore::Element
  @WithOpenFile(fout -> file)
    self.toWalker(fout)
  end
