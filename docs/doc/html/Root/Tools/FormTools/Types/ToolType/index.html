@Operation mergeParents():XCore::Element
  @For parent in parents do
    @For inheritedElement in parent.elements() do
      let findSeq = self.elements()->asSeq;
          findDone = false;
          findResult = null
      in declare(hidden,findSeq);
         declare(hidden,findDone);

         @While not findSeq->isEmpty and not findDone do
