NodeType

Overview

A node type names and describes the display structure for a class of nodes on a diagram. Node types may inherit which allows edges to be defined between multiple node types. Nodes are containers of display elements and as such define the layout of their contents.

@Operation new(x : Integer,y : Integer,tool : XCore::Element,diagram : XCore::Element):XCore::Element
  let node = self.newNode(x,y);
      env = displays->iterate(d env = NullEnv() |
              env.bind(d.name().toString(),d)) then
      selectionHandler = NodeType::mkSelectionHandler(tool,node);
      deselectionHandler = NodeType::mkDeselectionHandler(tool,node);
      clickHandler = NodeType::mkClicksHandler(tool,node);
      moveHandler = NodeType::mkMoveHandler(tool,node);
      resizeHandler = NodeType::mkResizeHandler(tool,node) then
      dia = tool.diagram()
  in if dia.hasCommandInterpreter()
     then let command = dia.commandInterpreter()
          in command.setPacketSize(30);
             command.stopRender(dia);
             try
               let result = node.setType(self);
                            node.setMinWidth(minWidth);
                            node.setMinHeight(minHeight);
                            node.setMaxWidth(maxWidth);
                            node.setMaxHeight(maxHeight);
                            @For display in displays do
                              display.new(node,node,Seq{name.toString()},tool,env)
                            end;
                            if hasPort
                            then node.addPortAssociation(node)
                            else false
                            end ;
                            node.new(diagram);
                            node.initDiagrams(tool);
                            self.setGlobalMenu(node,tool,Seq{self.name.toString()});
                            node.setSelectionHandler(selectionHandler);
                            node.setDeselectionHandler(deselectionHandler);
                            node.setClickHandler(clickHandler);
                            node.setMoveHandler(moveHandler);
                            node.setResizeHandler(resizeHandler);
                            diagram.add(node);
                            node.layout();
                            node
               in command.startRender(dia);
                  command.setPacketSize(1);
                  result
               end
             catch(exception)
               command.startRender(dia);
               command.setPacketSize(1);
               throw exception
             end
          end
     else
       node.setType(self);
       node.setMinWidth(minWidth);
       node.setMinHeight(minHeight);
       node.setMaxWidth(maxWidth);
       node.setMaxHeight(maxHeight);

         display.new(node,node,Seq{name.toString()},tool,env)

       else false

       node.setDeselectionHandler(deselectionHandler);

       diagram.add(node);
