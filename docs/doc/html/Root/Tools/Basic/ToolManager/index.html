ToolManager

Overview

A tool manager maintains associations between classes and a collection of XTool descriptors. Each tool descriptor manages a collection of tools.

@Operation saveXTO(element : XCore::Element):XCore::Element
  if not saving
  then 
    self.setSaving(true);
    let excludedNameSpaces = self.excludedNameSpaces(element)->asSeq;
        allTooledElements = self.tooledElements()->asSeq then
        nonRefTooledElements = Kernel_getNonReferencedElements(allTooledElements,excludedNameSpaces);
        T = Seq{} then
        tooledElementsForSaving = let walker = Walkers::FindBound(@Operation anonymous(x : XCore::Element):XCore::Element
                                                 nonRefTooledElements->includes(x)
                                               end,@Operation anonymous(x : XCore::Element):XCore::Element
                                                 x <> element andthen x.isReallyKindOf(NamedElement) andthen excludedNameSpaces->includes(x.owner())
                                               end)
                                  in walker.walk(element,null);
                                     walker.results->collect(x |
                                       x)
                                  end
    in @For e in tooledElementsForSaving do
         T := T + self.toolsFor(e)
       end;
       let savedTools = SavedTools(element,T,xmf.version)
       in savedTools.save()
       end
    end;
    self.setSaving(false)
  else false
  end 
end

@Operation toolDescriptor(c : Class,n : String):ToolDescriptor
  if descriptors.hasKey(c)
  then let nameSpace = descriptors.get(c)
       in if nameSpace.hasElement(n)
          then nameSpace.getElement(n)
          else self.error(formats("Cannot find a tool descriptor for ~S named ~S.",Seq{c.name(),n}))
          end 
       end
  else self.error(formats("Class ~S has not tool descriptors.",Seq{c.name()}))
  end 
end

                    then self.allToolsRecursive()

                    end

       else false
       end 
     end;

  end
