@Operation objectAttributes(object : Object):Seq(Attribute)
  let class = object.of() then
      caseObj = object then
      caseObjType = caseObj.of()
  in if caseObjType = DiagramTools::Diagrams::Edge
     then class.allAttributes()->excluding(a |
            a.name().toString() = "owner" or a.name().toString() = "commandInterpreter")->asSeq
     elseif caseObjType = DiagramTools::Diagrams::Node
     then class.allAttributes()->excluding(a |
            a.name().toString() = "owner" or a.name().toString() = "commandInterpreter")->asSeq
     else class.allAttributes()->asSeq
     end 
  end
