                              left = caseObj.left
                          in stack.put(frame + CODE,Seq{left | code->tail});

                          end

                          in self.failFrame := stack.ref(frame + CUT);

                          in self.pushFrame(stack.ref(frame + INDENT) + cols,stack.ref(frame + CUT),Seq{doc})
                          end

                     then let right = caseObj.right then
                              left = caseObj.left
                          in stack.put(frame + CODE,Seq{left,right} + code->tail)
                          end
