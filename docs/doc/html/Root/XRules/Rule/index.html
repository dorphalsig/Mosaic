@Operation add(element : XCore::Element):XCore::Element
  @TypeCase(element)
    RuleDef do
      let findSeq = ruleDefs->asSeq;
          findDone = false;
          findResult = null
      in declare(hidden,findSeq);
         declare(hidden,findDone);
         declare(hidden,findResult);
         @While not findSeq->isEmpty and not findDone do
           let ruleDef = findSeq->head then
               findElement = findSeq->head
           in declare(hidden,findElement);
              if ruleDef.arity() = element.arity()
              then 
                findDone := true;
                findResult := @For clause in element.clauses() do
                                ruleDef.add(clause)
                              end
              else findSeq := findSeq->tail
              end 
           end
         end;
         if not findDone
         then self.addToRuleDefs(element)
         else findResult
         end 
      end
    end
  end
end
